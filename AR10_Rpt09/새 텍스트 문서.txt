#include <Servo.h>
const int pinServo = 9;
const int pinButtonRight = 12;
const int pinButtonLeft = 13;

Servo servo;
int servoAngle = 90;

int rightButtonState = 0;
int leftButtonState = 0;
void setup() {
pinMode(pinButtonRight, INPUT);
pinMode(pinButtonLeft, INPUT);
servo.attach(pinServo);
servo.write(servoAngle);
}
void loop() {
  rightButtonState = digitalRead(pinButtonRight);
  leftButtonState = digitalRead(pinButtonLeft);
  
if (leftButtonState == 1) {
  if (servoAngle > 0) {
    servo.write(servoAngle);
    servoAngle--;
    delay(15);
    }
   }
if (rightButtonState == 1) {
  if (servoAngle < 180) {
    servo.write(servoAngle);
    servoAngle++;
    delay(15);
  }
 }
}
